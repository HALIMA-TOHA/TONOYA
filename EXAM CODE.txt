QUES.NO.2
---------

SET SERVEROUTPUT ON
DECLARE
V_ID EMPLOYEES.EMPLOYEE_ID%TYPE;
V_LNAME EMPLOYEES.LAST_NAME%TYPE;
V_SALARY NUMBER;
V_JOB_ID VARCHAR2(30);
BEGIN
  SELECT EMPLOYEE_ID,LAST_NAME,SALARY,JOB_ID
  INTO  V_ID,V_LNAME,V_SALARY,V_JOB_ID
  FROM EMPLOYEES
  WHERE EMPLOYEE_ID = 110;
  DBMS_OUTPUT.PUT_LINE('EMPLOYEE ID IS '||V_ID);
  DBMS_OUTPUT.PUT_LINE('LAST NAME IS '||V_LNAME);
  DBMS_OUTPUT.PUT_LINE('SALARY IS '||V_SALARY);
  DBMS_OUTPUT.PUT_LINE('JOB ID IS '||V_JOB_ID);
END;

SET SERVEROUTPUT ON
DECLARE
V_MARKS NUMBER :=0;
BEGIN
V_MARKS:=85;
CASE 
WHEN V_MARKS>=80 THEN
DBMS_OUTPUT.PUT_LINE('EXCELLENT');
WHEN V_MARKS>=50 THEN
DBMS_OUTPUT.PUT_LINE('GOOD');
WHEN V_MARKS>=40 THEN
DBMS_OUTPUT.PUT_LINE('AVERAGE');
ELSE
DBMS_OUTPUT.PUT_LINE('POOR');
END CASE;
END;
